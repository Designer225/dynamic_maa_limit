# In practice, most of the time titles aren't being transferred from one top realm to another.
# When they are though, propagating the effects upwards will be necessary.

	on_actions = {
		on_dynamic_maa_limit_update_propagate_upwards
	}
}

# no scope, used to update everyone so don't need propagation
on_game_start = {
	on_actions = {
		dynamic_maa_limit_yearly_global_pulse
		on_dynamic_maa_limit_update_no_scope
	}
}

# no scope, used to update everyone so don't need propagation
on_game_start_after_lobby = {
	on_actions = {
		dynamic_maa_limit_yearly_global_pulse
		on_dynamic_maa_limit_update_no_scope
	}
}

# only playable characters can have MAA regiments so this is enough
# also needs no propagation
on_yearly_playable = {
	on_actions = {
		on_dynamic_maa_limit_update
	}
}

# this scopes to everyone in order to update everyone
on_dynamic_maa_limit_update_no_scope = {
	effect = {
		every_ruler = {
			trigger_event = {
				on_action = on_dynamic_maa_limit_update
			}
		}
	}
}

yearly_global_pulse = {
	on_actions = {
		dynamic_maa_limit_yearly_global_pulse
		on_dynamic_maa_limit_update_no_scope
	}
}

dynamic_maa_limit_yearly_global_pulse = {
	effect = {
		set_global_variable = {
			name = additional_maa_cap_from_domain_size_multiplier
			value = additional_maa_cap_from_domain_size_mult
		}
		set_global_variable = {
			name = additional_maa_cap_from_realm_size_multiplier
			value = additional_maa_cap_from_realm_size_mult
		}
		set_global_variable = {
			name = additional_maa_cap_from_realm_size_soft_cap_value
			value = additional_maa_cap_from_realm_size_soft_cap
		}
		set_global_variable = {
			name = additional_maa_cap_from_realm_size_over_soft_cap_mult
			value = additional_maa_cap_from_realm_size_over_soft_cap_multiplier
		}
		set_global_variable = {
			name = additional_maa_cap_adventurers_addend
			value = additional_maa_cap_adventurers_add
		}
		set_global_variable = {
			name = additional_maa_cap_from_domicile_buildings_multiplier
			value = additional_maa_cap_from_domicile_buildings_mult
		}
		set_global_variable = {
			name = additional_maa_cap_from_adventurer_camp_buildings_multiplier
			value = additional_maa_cap_from_adventurer_camp_buildings_mult
		}
	}
}

# updates only one character
on_dynamic_maa_limit_update = {
	## intent
	# 	- Landed rulers get more regiment cap from lands. Nuff said.
	#		- Specifically, more from each holding than from each county. Significantly so.
	#		- Idea is to make larger realms more powerful without being overly so
	#	- Unlanded rulers obviously can't get regiment cap from land, and need to source it elsewhere
	#		- Unlanded rulers have domiciles and can instead get regiment cap from each building in the domicile
	#	- Gotchas: recent landless government types
	#		- Administrative realms (admin, meritocratic (khanate), celestial, ritsuryo), Soryo realms, and Nomads
	#			- Landed rulers can have domiciles (as house leaders) and not (as non-cadet relatives of house leaders)
	#				- house leaders will get modifier from domiciles only
	#				- domicile-less will get the 'adventurer' modifier only ('adventurer' because it only applies to them now)
	#				- This is on top of the land modifiers (they ARE landed after all)
	#					- Note that their land modifiers apply to TITLE MAAs if they have them
	#						- Admins can have them if they belong to a tier that allows them. It's usually duchy or higher (county in some cases) so it's mostly fine
	effect = {
		remove_all_character_modifier_instances = additional_maa_cap_from_domain_size
		remove_all_character_modifier_instances = additional_maa_cap_from_realm_size
		remove_all_character_modifier_instances = additional_title_maa_cap_from_domain_size
		remove_all_character_modifier_instances = additional_title_maa_cap_from_realm_size
		remove_all_character_modifier_instances = additional_maa_cap_adventurers
		remove_all_character_modifier_instances = additional_maa_cap_from_domicile_buildings
		remove_all_character_modifier_instances = additional_maa_cap_from_adventurer_buildings
		if = { # landed admin types, feudal, clan, tribal, landed nomads, mandala, wanua, landed soryo
			limit = { is_landed = yes }
			if = {
				limit = { government_has_flag = government_has_title_men_at_arms }
				if = {
					limit = { NOT = { has_game_rule = no_maa_per_domain_size } }
					add_character_modifier = additional_title_maa_cap_from_domain_size
				}
				if = {
					limit = { NOT = { has_game_rule = no_maa_per_realm_size } }
					add_character_modifier = additional_title_maa_cap_from_realm_size
				}
			}
			else = {
				if = {
					limit = { NOT = { has_game_rule = no_maa_per_domain_size } }
					add_character_modifier = additional_maa_cap_from_domain_size
				}
				if = {
					limit = { NOT = { has_game_rule = no_maa_per_realm_size } }
					add_character_modifier = additional_maa_cap_from_realm_size
				}
			}
		}

		if = { # all admin types and soryo
			limit = {
				government_has_flag = government_uses_domicile_but_not_adventurer
				is_nomad = no
			}
			if = {
				limit = { has_domicile = yes }
				add_character_modifier = additional_maa_cap_from_domicile_buildings
			}
			else = {
				add_character_modifier = additional_maa_cap_adventurers
			}
		}
		else_if = { # nomads and adventurers
			limit = {
				OR = {
					is_landless_adventurer = yes
					is_nomad = yes
				}
			}
			if = {
				limit = { has_domicile = yes }
				add_character_modifier = additional_maa_cap_from_adventurer_buildings
			}
			else = {
				add_character_modifier = additional_maa_cap_adventurers
			}
		}
	}
}

# updates a character and all of their lieges, if any
on_dynamic_maa_limit_update_propagate_upwards = {
	effect = {
		# update current character
		trigger_event = {
			on_action = on_dynamic_maa_limit_update
		}
		# now update the immediate liege, if any, until there is no more liege
		if = {
			limit = {
				exists = liege
				liege != this
			}
			liege = {
				trigger_event = {
					on_action = on_dynamic_maa_limit_update_propagate_upwards
				}
			}
		}
	}
}
